建表语句：

CREATE TABLE `bs_admin_user` (
  `id` bigint(64) NOT NULL COMMENT '主键ID',
  `user_id` varchar(64) NOT NULL COMMENT '用户Id',
  `user_name` varchar(45) NOT NULL COMMENT '用户名',
  `account` varchar(64) NOT NULL,
  `password` varchar(45) NOT NULL COMMENT '密码',
  `mobile` varchar(45) NOT NULL COMMENT '手机号',
  `email` varchar(45) DEFAULT NULL COMMENT '邮箱地址',
  `contact_name` varchar(45) DEFAULT NULL COMMENT '联系人姓名',
  `contact_mobile` varchar(45) DEFAULT NULL COMMENT '联系人手机号',
  `contact_email` varchar(45) DEFAULT NULL COMMENT '联系人邮箱',
  `wx_id` varchar(45) DEFAULT NULL COMMENT '微信Id',
  `qq` varchar(45) DEFAULT NULL COMMENT 'qq号',
  `status` int(11) NOT NULL DEFAULT '1' COMMENT '状态信息:1有效数据、2暂存数据、0无效',
  `update_time` datetime NOT NULL COMMENT '最后更新时间',
  `updator` varchar(45) NOT NULL COMMENT '更新人用户code',
  `create_time` datetime NOT NULL,
  `creator` varchar(45) NOT NULL COMMENT '创建者用户code',
  PRIMARY KEY (`id`),
  KEY `user_id_index` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='用户表';


CREATE TABLE `bs_admin_department` (
  `id` bigint(64) NOT NULL COMMENT '主键ID',
  `department_id` varchar(64) NOT NULL COMMENT '部门Id',
  `name` varchar(64) NOT NULL COMMENT '部门名称',
  `parent_department_id` varchar(64) NOT NULL DEFAULT '0' COMMENT '上级部门Id.最高级部门上线Id为0',
  `status` int(11) NOT NULL DEFAULT '1' COMMENT '状态信息:1有效数据、2暂存数据、0无效',
  `update_time` datetime NOT NULL COMMENT '最后更新时间',
  `updator` varchar(45) NOT NULL COMMENT '更新人用户code',
  `create_time` datetime NOT NULL,
  `creator` varchar(45) NOT NULL COMMENT '创建者用户code',
  PRIMARY KEY (`id`),
  KEY `department_id` (`department_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='部门信息表';



CREATE TABLE `bs_admin_permit` (
  `id` bigint(64) NOT NULL COMMENT '主键ID',
  `permit_id` varchar(64) NOT NULL COMMENT '权限Id',
  `text` varchar(45) NOT NULL COMMENT '权限名称',
  `platform_id` varchar(64) NOT NULL DEFAULT 'system' COMMENT '平台Id',
  `data` varchar(100) DEFAULT NULL COMMENT '菜单数据存放\n',
  `permit_type` int(11) NOT NULL DEFAULT '1' COMMENT '权限类型:1菜单、2功能按钮',
  `parent_permit_id` varchar(64) NOT NULL DEFAULT '0' COMMENT '父级权限Id,顶级菜单权限父级Id为0、功能级权限父级Id为其上级菜单权限Id',
  `icon` varchar(45) DEFAULT NULL COMMENT '菜单icon',
  `extremity` int(11) NOT NULL COMMENT '菜单权限是否为最后一级 0:否、1是',
  `sort_no` int(11) NOT NULL COMMENT '排序字段',
  `url` varchar(100) NOT NULL COMMENT '权限地址',
  `status` int(11) NOT NULL DEFAULT '1' COMMENT '状态信息:1有效数据、2暂存数据、0无效',
  `update_time` datetime NOT NULL COMMENT '最后更新时间',
  `updator` varchar(45) NOT NULL COMMENT '更新人用户code',
  `create_time` datetime NOT NULL,
  `creator` varchar(45) NOT NULL COMMENT '创建者用户code',
  PRIMARY KEY (`id`),
  KEY `permit_id_index` (`permit_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='权限表';
